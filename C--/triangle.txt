const int maxa = 2e3 + 5;
int n, m, k;
ll a[maxa][maxa], ngang[maxa][maxa], cheo[maxa][maxa], doc[maxa][maxa], d[maxa][maxa];
ll ans = -oo;

    for (int i = 2; i <= n; i++) {

            ans = max(ans, d[i][j]);
        }
    }
