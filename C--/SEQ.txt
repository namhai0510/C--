#define ll long long
#define ull unsigned long long
const int maxn = 1e7 + 5;
const int maxa = 1e3 + 5;
ull n,m,res;
ull nhan(ull a, ull b, ull mod) {
    if (b == 0) return 0;
    ull t = nhan(a, b / 2, mod);      
    t = (t + t) % mod; 
    if (b % 2 == 1) t = (t + a) % mod;
    return t;
} 
ull power(ull a, ull b, ull mod) {
    if (b == 0) return 1;
    ull t = power(a, b / 2, mod);         
    t = nhan(t, t, mod) % mod; 
    if (b % 2 == 1) t = nhan(t , a, mod);
    return t;
} 
void Input() {
    cin >> n >> m;
}
void Solve() {
    ull tt = power(3, n, m);
    res = (n + 1)%m;
    res = nhan(res, tt, m);
}
void Output() {
    cout << res % m;
}
